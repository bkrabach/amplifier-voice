# Realtime 1.5 Upgrade Implementation Plan

> **Execution:** Use the subagent-driven-development workflow to implement this plan.

**Goal:** Upgrade amplifier-voice from `gpt-realtime` to `gpt-realtime-1.5` with a server-side sideband control plane, async function calling (delegate + dispatch), and retention ratio truncation.

**Architecture:** The browser becomes a pure audio transport. A new `VoiceSideband` class opens a WebSocket to the same OpenAI session the browser uses, giving the server full control over tool execution, result injection, and context management. Two tools — synchronous `delegate` and fire-and-forget `dispatch` — replace the single client-routed tool call path.

**Tech Stack:** Python 3.11+ / FastAPI / httpx / websockets (backend), TypeScript / React 18 (frontend), pytest + pytest-asyncio (tests)

**Design doc:** `amplifier-voice/docs/plans/2026-03-10-realtime-1.5-upgrade-design.md`

```python
"""Voice Sideband — server-side WebSocket connection to an OpenAI RT session.

After the browser establishes a WebRTC connection with OpenAI, the server
opens a sideband WebSocket to the SAME session. This connection:

- Monitors for tool calls (response.function_call_arguments.done)
- Executes them via the AmplifierBridge
- Injects results back (conversation.item.create + response.create)
- Does NOT handle audio — that flows directly via WebRTC
"""

from __future__ import annotations

import asyncio
import json
import logging
from typing import Any, TYPE_CHECKING

import websockets

if TYPE_CHECKING:
    from .amplifier_bridge import AmplifierBridge

logger = logging.getLogger(__name__)

OPENAI_REALTIME_WS = "wss://api.openai.com/v1/realtime"
TOOL_CALL_EVENT = "response.function_call_arguments.done"


class VoiceSideband:
    """Manages a sideband WebSocket connection to an OpenAI Realtime session."""

    def __init__(
        self,
        bridge: AmplifierBridge,
        api_key: str,
        call_id: str,
    ) -> None:
        self.bridge = bridge
        self._api_key = api_key
        self.call_id = call_id
        self._ws: Any = None
        self._listen_task: asyncio.Task | None = None
        self.is_connected = False

    def _build_ws_url(self) -> str:
        """Build the WebSocket URL for the sideband connection."""
        return f"{OPENAI_REALTIME_WS}?call_id={self.call_id}"

    async def connect(self) -> None:
        """Open the sideband WebSocket connection and start listening."""
        url = self._build_ws_url()
        headers = {"Authorization": f"Bearer {self._api_key}"}

        logger.info("Sideband connecting to call %s", self.call_id)
        try:
            self._ws = await websockets.connect(url, additional_headers=headers)
            self.is_connected = True
            logger.info("Sideband connected to call %s", self.call_id)
            self._listen_task = asyncio.create_task(self._listen_loop())
        except Exception as e:
            logger.error("Sideband connection failed for call %s: %s", self.call_id, e)
            self.is_connected = False

    async def _listen_loop(self) -> None:
        """Listen for events on the sideband WebSocket."""
        try:
            async for raw_msg in self._ws:
                try:
                    event = json.loads(raw_msg)
                except json.JSONDecodeError:
                    logger.warning("Sideband: non-JSON message (%d bytes)", len(raw_msg))
                    continue
                await self._handle_event(event)
        except asyncio.CancelledError:
            pass
        except Exception:
            logger.exception("Sideband listen loop error")
        finally:
            self.is_connected = False

    async def _handle_event(self, event: dict[str, Any]) -> None:
        """Handle a single event from the sideband WebSocket."""
        event_type = event.get("type", "")

        if event_type == TOOL_CALL_EVENT:
            await self._handle_tool_call(event)
        elif event_type == "error":
            error_msg = event.get("error", {}).get("message", "Unknown error")
            logger.error("Sideband received error from OpenAI: %s", error_msg)
        else:
            logger.debug("Sideband ignoring event: %s", event_type)

    async def _handle_tool_call(self, event: dict[str, Any]) -> None:
        """Execute a tool call and inject the result back into the session."""
        call_id = event.get("call_id", "")
        name = event.get("name", "")
        raw_arguments = event.get("arguments", "{}")

        logger.info("Sideband tool call: %s (call_id=%s)", name, call_id)

        try:
            arguments = json.loads(raw_arguments) if isinstance(raw_arguments, str) else raw_arguments
        except json.JSONDecodeError:
            logger.error("Failed to parse tool arguments for %s: %s", name, raw_arguments)
            arguments = {}

        try:
            result = await self.bridge.execute_tool(name, arguments)

            # Format output string
            if result.success:
                output = result.output if isinstance(result.output, str) else json.dumps(result.output)
            else:
                output = json.dumps({"error": result.error or "Unknown error"})

            # Send result back
            item_create = {
                "type": "conversation.item.create",
                "item": {
                    "type": "function_call_output",
                    "call_id": call_id,
                    "output": output,
                },
            }
            await self._ws.send(json.dumps(item_create))

            # Trigger model to respond
            response_create = {"type": "response.create"}
            await self._ws.send(json.dumps(response_create))

            logger.info("Sideband tool result injected for call_id=%s", call_id)
        except Exception:
            logger.exception("Sideband tool call failed: %s (call_id=%s)", name, call_id)

    async def inject_result(
        self,
        call_id: str,
        output: str,
        instructions: str = "A previously dispatched task just completed. Summarize the result concisely for the user.",
    ) -> None:
        """Inject an async tool result into the session.

        Used by dispatch background tasks to push results after they complete.
        No-op if not connected.
        """
        if not self.is_connected or not self._ws:
            return

        logger.info("Sideband injecting async result for call_id=%s", call_id)

        item_create = {
            "type": "conversation.item.create",
            "item": {
                "type": "function_call_output",
                "call_id": call_id,
                "output": output,
            },
        }

        response_create = {
            "type": "response.create",
            "response": {
                "instructions": instructions,
            },
        }

        try:
            await self._ws.send(json.dumps(item_create))
            await self._ws.send(json.dumps(response_create))
        except Exception:
            logger.exception("Sideband inject_result failed for call_id=%s", call_id)
            self.is_connected = False

    async def send_session_update(self, session_config: dict[str, Any]) -> None:
        """Send a session.update message through the sideband.

        Used for truncation config, instruction updates, etc.
        No-op if not connected.
        """
        if not self.is_connected or not self._ws:
            return

        msg = {
            "type": "session.update",
            "session": session_config,
        }

        try:
            await self._ws.send(json.dumps(msg))
            logger.info("Sideband sent session.update")
        except Exception:
            logger.exception("Sideband send_session_update failed")
            self.is_connected = False

    async def disconnect(self) -> None:
        """Close the sideband connection."""
        if self._listen_task:
            self._listen_task.cancel()
            try:
                await self._listen_task
            except asyncio.CancelledError:
                pass
            self._listen_task = None
        if self._ws:
            await self._ws.close()
            self._ws = None
        self.is_connected = False
        logger.info("Sideband disconnected from call %s", self.call_id)
```

    # Location: https://api.openai.com/v1/realtime/calls/{call_id}
    call_id = ""

        logger.info(f"SDP exchange successful, call_id={call_id}")
    else:
        logger.warning("SDP exchange successful but no Location header — call_id unavailable")

    return {"sdp": sdp_content, "call_id": call_id}

        Called by the client after WebRTC connection is established.
        The server opens a server-side WebSocket to the SAME OpenAI session.

        Body:

        - ephemeral_key: str - from /session response

            )

            )

        if call_id in _sideband_registry:

        sideband = VoiceSideband(
            bridge=_amplifier_bridge,
            api_key=ephemeral_key,
            call_id=call_id,
        )

        await sideband.connect()

        if not sideband.is_connected:
            raise HTTPException(
                status_code=502, detail="Failed to connect sideband WebSocket"
            )

async def run_agent_job(
    sideband: VoiceSideband,
    call_id: str,
    agent: str,
    instruction: str,
    bridge: "AmplifierBridge",
) -> None:
    """Run a background agent job and inject the result via sideband.

    This is spawned as an asyncio task for dispatch (fire-and-forget) calls.
    The lock is acquired only during the actual execute() call, then released
    so delegate calls can interleave.

    Args:
        sideband: The active sideband connection for this session.
        call_id: The original function call ID from OpenAI.
        agent: Agent name (e.g. "foundation:explorer").
        instruction: What the agent should do.
        bridge: AmplifierBridge instance for tool execution.
    """
    logger.info("Background job started: agent=%s call_id=%s", agent, call_id)

    try:
        # Acquire lock only during execution, not while waiting
        async with execution_lock:
            result = await bridge.execute_tool(
                "delegate",
                {"agent": agent, "instruction": instruction},
            )

        if result.success:
            output = result.output if isinstance(result.output, str) else json.dumps(result.output)
        else:
            output = f"Error: {result.error or 'Unknown error'}"

        await sideband.inject_result(call_id=call_id, output=output)
        logger.info("Background job completed: agent=%s call_id=%s", agent, call_id)

    except Exception as e:
        logger.exception("Background job failed: agent=%s call_id=%s", agent, call_id)
        error_output = f"Background task failed: {e}"
        await sideband.inject_result(call_id=call_id, output=error_output)

        """Route a tool call — delegate runs sync, dispatch runs async."""
        call_id = event.get("call_id", "")
        name = event.get("name", "")
        raw_arguments = event.get("arguments", "{}")

        logger.info("Sideband tool call: %s (call_id=%s)", name, call_id)

        try:
            arguments = json.loads(raw_arguments) if isinstance(raw_arguments, str) else raw_arguments
        except json.JSONDecodeError:
            logger.error("Failed to parse tool arguments for %s: %s", name, raw_arguments)
            arguments = {}

        if name == "dispatch":
            await self._handle_dispatch(call_id, arguments)
        elif name == "cancel_current_task":
            await self._handle_cancel(call_id, arguments)
        else:
            # delegate and all other tools — synchronous execution
            await self._handle_delegate(call_id, name, arguments)

    async def _handle_delegate(self, call_id: str, name: str, arguments: dict[str, Any]) -> None:
        """Execute a tool call synchronously and inject the result."""
        try:
            async with execution_lock:
                result = await self.bridge.execute_tool(name, arguments)

            # Format output string
            if result.success:
                output = result.output if isinstance(result.output, str) else json.dumps(result.output)
            else:
                output = json.dumps({"error": result.error or "Unknown error"})

            # Send result back
            item_create = {
                "type": "conversation.item.create",
                "item": {
                    "type": "function_call_output",
                    "call_id": call_id,
                    "output": output,
                },
            }
            await self._ws.send(json.dumps(item_create))

            # Trigger model to respond
            response_create = {"type": "response.create"}
            await self._ws.send(json.dumps(response_create))

            logger.info("Sideband delegate result injected for call_id=%s", call_id)
        except Exception:
            logger.exception("Sideband delegate call failed: %s (call_id=%s)", name, call_id)

    async def _handle_dispatch(self, call_id: str, arguments: dict[str, Any]) -> None:
        """Handle dispatch — immediate ack, then spawn background task."""
        agent = arguments.get("agent", "unknown")
        instruction = arguments.get("instruction", "")

        logger.info("Sideband dispatch: agent=%s call_id=%s", agent, call_id)

        # Immediate synthetic ack
        ack_output = f"Dispatched to {agent}. You'll be notified when complete."
        item_create = {
            "type": "conversation.item.create",
            "item": {

                "call_id": call_id,
                "output": ack_output,
            },
        }

        asyncio.create_task(

                agent=agent,
                instruction=instruction,
                bridge=self.bridge,
            )

        """Handle cancel_current_task — routes to bridge cancellation."""
        try:
            result = await self.bridge.execute_tool("cancel_current_task", arguments)
            output = result.output if isinstance(result.output, str) else json.dumps(result.output)

            item_create = {
                "type": "conversation.item.create",
                "item": {
                    "type": "function_call_output",
                    "call_id": call_id,
                    "output": output,
                },
            }
            await self._ws.send(json.dumps(item_create))
            await self._ws.send(json.dumps({"type": "response.create"}))
        except Exception:
            logger.exception("Sideband cancel failed (call_id=%s)", call_id)
